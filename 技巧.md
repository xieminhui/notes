### 1.数组快速去重
+ es6
```
var arr = new Set();
var temp = [1,2,1,1,3,3,5,5,6];
temp.map(function(x){
	arr.add(x);		
})
console.log(arr)//{1, 2, 3, 5, 6}

var temp = new Set([1,2,1,1,3,3,5,5,6]);//{1, 2, 3, 5, 6}
```
+ es5 
```
    var temp = [1,2,1,1,3,3,5,5,6];
    var obj ={};
    temp.map(function(x){
    	obj[x] = 0;
    })
    for(var key in obj){
    	console.log(key);
    }
    //时间复杂度O(n)
```
